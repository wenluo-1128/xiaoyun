  <script>
    // 交互：打印与复制坐标
    const coordText = "112.183509, 37.204136";
    const copyBtn = document.getElementById("copyBtn");
    const printBtn = document.getElementById("printBtn");

    // 打印功能
    printBtn.addEventListener("click", () => window.print());

    // 复制坐标功能
    async function copyCoord() {
      try {
        if (navigator.clipboard && window.isSecureContext) {
          await navigator.clipboard.writeText(coordText);
        } else {
          // 降级方案：创建临时输入框
          const input = document.createElement("input");
          input.value = coordText;
          document.body.appendChild(input);
          input.select();
          document.execCommand("copy");
          document.body.removeChild(input);
        }
        // 显示复制成功状态
        copyBtn.textC";
        setTimeout(() => copyBtn.textC", 1200);
      } catch (err) {
        console.warn("复制失败:", err);
        // 手动选中文本提示用户
        const range = document.createRange();
        range.selectNodeContents(document.getElementById("coordText"));
        const sel = window.getSelection();
        sel.removeAllRanges();
        sel.addRange(range);
        alert("复制失败，请手动复制坐标：" + coordText);
      }
    }

    copyBtn.addEventListener("click", copyCoord);
  </script>
